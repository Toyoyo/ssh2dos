226 /*
227  * Structure for query header. The order of the fields is machine- and
228  * compiler-dependent, depending on the byte/bit order and the layout
229  * of bit fields. We use bit fields only in int variables, as this
230  * is all ANSI requires. This requires a somewhat confusing rearrangement.
231  */
232 
233 typedef struct {
234  unsigned id : 16; /* query identification number */
235 
236  /* fields in third byte */
237  unsigned rd : 1; /* recursion desired */
238  unsigned tc : 1; /* truncated message */
239  unsigned aa : 1; /* authoritative answer */
240  unsigned opcode : 4; /* purpose of message */
241  unsigned qr : 1; /* response flag */
242 
243  /* fields in fourth byte */
244  unsigned rcode : 4; /* response code */
245  unsigned cd : 1; /* checking disabled by resolver */
246  unsigned ad : 1; /* authentic data from named */
247  unsigned unused : 1; /* unused bits (MBZ as of 4.9.3a3) */
248  unsigned ra : 1; /* recursion available */
249 
250  /* remaining bytes */
251  unsigned qdcount : 16; /* number of question entries */
252  unsigned ancount : 16; /* number of answer entries */
253  unsigned nscount : 16; /* number of authority entries */
254  unsigned arcount : 16; /* number of resource entries */
255  } HEADER;

286 /*
287  * Inline versions of get/put short/long. Pointer is advanced.
288  *
289  * These macros demonstrate the property of C whereby it can be
290  * portable or it can be elegant but rarely both.
291  */
292 #define GETSHORT(s, cp) do { \
293  register u_char *t_cp = (u_char *)(cp); \
294  (s) = ((u_short)t_cp[0] << 8) \
295  | ((u_short)t_cp[1]); \
296  (cp) += INT16SZ; \
297  } while (0)
298 
299 #define GETLONG(l, cp) do { \
300  register u_char *t_cp = (u_char *)(cp); \
301  (l) = ((u_long)t_cp[0] << 24) \
302  | ((u_long)t_cp[1] << 16) \
303  | ((u_long)t_cp[2] << 8) \
304  | ((u_long)t_cp[3]); \
305  (cp) += INT32SZ; \
306  } while (0)

308 #define PUTSHORT(s, cp) do { \
309  register u_short t_s = (u_short)(s); \
310  register u_char *t_cp = (u_char *)(cp); \
311  *t_cp++ = (u_char) (t_s >> 8); \
312  *t_cp = (u_char) t_s; \
