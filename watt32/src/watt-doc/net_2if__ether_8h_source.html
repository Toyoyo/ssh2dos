97 /*
98  * Macro to map an IP multicast address to an Ethernet multicast address.
99  * The high-order 25 bits of the Ethernet address are statically assigned,
100  * and the low-order 23 bits are taken from the low end of the IP address.
101  */
102 #define ETHER_MAP_IP_MULTICAST(ipaddr, enaddr) \
103  /* struct in_addr *ipaddr; */ \
104  /* u_char enaddr[6]; */ \
105  { \
106  (enaddr)[0] = 0x01; \
107  (enaddr)[1] = 0x00; \
108  (enaddr)[2] = 0x5e; \
109  (enaddr)[3] = ((u_char *)ipaddr)[1] & 0x7f; \
110  (enaddr)[4] = ((u_char *)ipaddr)[2]; \
111  (enaddr)[5] = ((u_char *)ipaddr)[3]; \
112  }

170 /*
171  * Ethernet multicast address structure. There is one of these for each
172  * multicast address or range of multicast addresses that we are supposed
173  * to listen to on a particular interface. They are kept in a linked list,
174  * rooted in the interface's arpcom structure. (This really has nothing to
175  * do with ARP, or with the Internet address family, but this appears to be
176  * the minimally-disrupting place to put it.)
177  */
178 struct ether_multi {
179  u_char enm_addrlo[6]; /* low or only address of range */
180  u_char enm_addrhi[6]; /* high or only address of range */
181  struct arpcom *enm_ac; /* back pointer to arpcom */
182  u_int enm_refcount; /* no. claims to this addr/range */
183  struct ether_multi *enm_next; /* ptr to next ether_multi */
184  };

198 /*
199  * Macro for looking up the ether_multi record for a given range of Ethernet
200  * multicast addresses connected to a given arpcom structure. If no matching
201  * record is found, "enm" returns NULL.
202  */
203 #define ETHER_LOOKUP_MULTI(addrlo, addrhi, ac, enm) \
204  /* u_char addrlo[6]; */ \
205  /* u_char addrhi[6]; */ \
206  /* struct arpcom *ac; */ \
207  /* struct ether_multi *enm; */ \
208 { \
209  for ((enm) = (ac)->ac_multiaddrs; \
210  (enm) != NULL && \
211  (bcmp((enm)->enm_addrlo, (addrlo), 6) != 0 || \
212  bcmp((enm)->enm_addrhi, (addrhi), 6) != 0); \
213  (enm) = (enm)->enm_next); \
214 }
215 
216 /*
217  * Macro to step through all of the ether_multi records, one at a time.
218  * The current position is remembered in "step", which the caller must
219  * provide. ETHER_FIRST_MULTI(), below, must be called to initialize "step"
220  * and get the first record. Both macros return a NULL "enm" when there
221  * are no remaining records.
222  */
223 #define ETHER_NEXT_MULTI(step, enm) \
224  /* struct ether_multistep step; */ \
225  /* struct ether_multi *enm; */ \
226 { \
227  if (((enm) = (step).e_enm) != NULL) \
228  (step).e_enm = (enm)->enm_next; \
229 }
230 
231 #define ETHER_FIRST_MULTI(step, ac, enm) \
232  /* struct ether_multistep step; */ \
233  /* struct arpcom *ac; */ \
234  /* struct ether_multi *enm; */ \
235 { \
236  (step).e_enm = (ac)->ac_multiaddrs; \
237  ETHER_NEXT_MULTI((step), (enm)); \
