44 {

50  {

53  return (-1);
54  }
55 
56  if (socket->so_type != SOCK_STREAM)
57  {
58  SOCK_DEBUGF ((", EOPNOTSUPP"));

84  /* A listen socket should never time out unless

86  * accept().
87  */
88  socket->timeout = 0;
