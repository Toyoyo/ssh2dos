44 /*
45  * TCP FSM state definitions.
46  * Per RFC793, September, 1981.
47  */
48 
49 #define TCP_NSTATES 11
50 
51 #define TCPS_CLOSED 0 /* closed */
52 #define TCPS_LISTEN 1 /* listening for connection */
53 #define TCPS_SYN_SENT 2 /* active, have sent syn */
54 #define TCPS_SYN_RECEIVED 3 /* have send and received syn */
55 /* states < TCPS_ESTABLISHED are those where connections not established */
56 #define TCPS_ESTABLISHED 4 /* established */
57 #define TCPS_CLOSE_WAIT 5 /* rcvd fin, waiting for close */
58 /* states > TCPS_CLOSE_WAIT are those where user has closed */
59 #define TCPS_FIN_WAIT_1 6 /* have closed, sent fin */
60 #define TCPS_CLOSING 7 /* closed xchd FIN; await FIN ACK */
61 #define TCPS_LAST_ACK 8 /* had fin and close; await FIN ACK */
62 /* states > TCPS_CLOSE_WAIT && < TCPS_FIN_WAIT_2 await ACK of FIN */
63 #define TCPS_FIN_WAIT_2 9 /* have closed, fin is acked */
64 #define TCPS_TIME_WAIT 10 /* in 2*msl quiet wait after close */
65 
66 #define TCPS_HAVERCVDSYN(s) ((s) >= TCPS_SYN_RECEIVED)
67 #define TCPS_HAVEESTABLISHED(s) ((s) >= TCPS_ESTABLISHED)
68 #define TCPS_HAVERCVDFIN(s) ((s) >= TCPS_TIME_WAIT)
