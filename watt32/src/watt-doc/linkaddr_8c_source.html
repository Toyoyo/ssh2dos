57 void W32_CALL link_addr (const char *addr, struct sockaddr_dl *sdl)
58 {
59  char *cp = sdl->sdl_data;
60  char *cplim = sdl->sdl_len + (char*) sdl;
61  int Byte = 0;
62  int state = NAMING;
63  int New = 0;
64 
65  memset (&sdl->sdl_family, 0, sdl->sdl_len - 1);
66  sdl->sdl_family = AF_LINK;
67 
68  do
69  {
70  state &= ~LETTER;
71  if (*addr >= '0' && *addr <= '9')
72  New = *addr - '0';
73 
74  else if (*addr >= 'a' && *addr <= 'f')
75  New = *addr - 'a' + 10;
76 
77  else if (*addr >= 'A' && *addr <= 'F')
78  New = *addr - 'A' + 10;
79 
80  else if (*addr == 0)
81  state |= END;
82 
83  else if (state == NAMING &&
84  ((*addr >= 'A' && *addr <= 'Z') ||
85  (*addr >= 'a' && *addr <= 'z')))
86  state |= LETTER;
87 
88  else
89  state |= DELIM;
90 
91  addr++;
92  switch (state)
93  {
94  case NAMING | DIGIT:
95  case NAMING | LETTER:
96  *cp++ = addr[-1];
97  continue;
98  case NAMING | DELIM:
99  state = RESET;
100  sdl->sdl_nlen = (u_char)(cp - sdl->sdl_data);
101  continue;
102  case GOTTWO | DIGIT:
103  *cp++ = Byte;
104  /* FALLTHROUGH */
105  case RESET | DIGIT:
106  state = GOTONE;
107  Byte = New;
108  continue;
109  case GOTONE | DIGIT:
110  state = GOTTWO;
111  Byte = New + (Byte << 4);
112  continue;
113  default: /* | DELIM */
114  state = RESET;
115  *cp++ = Byte;
116  Byte = 0;
117  continue;
118  case GOTONE | END:
119  case GOTTWO | END:
120  *cp++ = Byte;
121  /* FALLTHROUGH */
122  case RESET | END:
123  break;
124  }
125  break;
126  }
127  while (cp < cplim);
128 
129  sdl->sdl_alen = (u_char)(cp - LLADDR(sdl));
130  New = (int) (cp - (char*)sdl);
131  if (New > (int)sizeof(*sdl))
132  sdl->sdl_len = New;
133 }

135 char * W32_CALL link_ntoa (const struct sockaddr_dl *sdl)
136 {
137  static char obuf[64];

139  const BYTE *in = (const BYTE*) LLADDR (sdl);
140  const BYTE *inlim = in + sdl->sdl_alen;
141  BOOL firsttime = TRUE;
142 
143  if (sdl->sdl_nlen)
144  {
145  memcpy (obuf, sdl->sdl_data, sdl->sdl_nlen);
146  out += sdl->sdl_nlen;
147  if (sdl->sdl_alen)
148  *out++ = ':';
149  }
150  while (in < inlim && out < obuf+sizeof(obuf)-3)
151  {
152  int Byte = *in++;
153

158  {

161  }

164  }
