732 /* Routing table internal structure:
733  * \verbatim
734  *
735  * Index pointers Route cache table
736  * -------------- -----------------
737  *
738  * (top_of_cache) ->
739  * (= top_pending) ----------------------
740  * | ...
741  * | PENDING ROUTE ENTRIES (ARPing gateway - no reply yet)
742  * first_pending -> | ... grow downwards (into free slots)
743  * (= top_free) ----------------------
744  * | ...
745  * | ...
746  * | FREE SLOTS
747  * | ...
748  * first_free ----> | ...
749  * (= top_dynamic) ----------------------
750  * | ... grow upwards (into free slots)
751  * | ...
752  * | "DYNAMIC" ROUTE ENTRIES
753  * first_dynamic -> | ...
754  * ----------------------
755  * \endverbatim
756  *
757  * \note
758  * - "Top" means last entry + 1.
759  * - The entries inside each section are not ordered in any way.
760  * - The route cache only holds entries of hosts that are OUTSIDE of our LAN.
761  * - The entries time-out when the gateways ARP cache entry times out.
762  *
763  * \note
764  * Although we don't support multiple physical interfaces, we support
765  * multiple gateways connected to our interface.
766  */

974

976  */
977  for (i = route_first_pending; i < ROUTE_TOP_PENDING; i++)
978  if (route_list[i].host_ip == ip)
979  return (TRUE); /* Already here */
980

982  if (route_list[i].host_ip == ip)

991  }

994  */

996  first_gate_mask = 0;

1047 static void route_check_timeouts (BOOL check_dynamic_entries)
1048 {
1049  struct route_entry *re, temp;
1050  int i, j;
1051 
1052  /* Check our pending entries
1053  */
1054  for (i = route_first_pending; i < ROUTE_TOP_PENDING; i++)
1055  {
1056  re = route_list + i;
1057 
1058  /* Was the ARP lookup able to resolve the gateway IP?
1059  */
1060  if (LAN_lookup(re->gate_ip, NULL))
1061  {
1062  /* Success - move route entry from pending to dynamic list
1063  */
1064  temp = *re; /* Make a copy so we can safely delete the pending entry */
1065  if (i > route_first_pending)
1066  route_move_entry (i--, route_first_pending); /* fill hole */
1067  /* (i-- to "re"check new current entry) */
1068 
1069  temp.flags &= ~ROUTE_FLG_PENDING;
1070  ++route_first_pending; /* remove from pending list */
1071  route_list [route_top_dynamic] = temp; /* add to dynamic list */
1072  ++route_top_dynamic;
1073  }
1074  /* Is the ARP lookup still pending? -> Keep waiting
1075  */
1076  else if (LAN_lookup_pending(re->gate_ip))
1077  {
1078  /* Do nothing */
1079  }
1080  /* The ARP lookup failed, we try the next possible gateway
1081  */
1082  else
1083  {
1084  /* Find the gateway that was tried last (the one that just timed out)
1085  */
1086  BOOL found_last_gw = FALSE;
1087  BOOL found_next_gw = FALSE;
1088 
1089  for (j = 0; j < gate_top; j++)
1090  {
1091  if (gate_list[j].gate_ip != re->gate_ip)
1092  continue;
1093  found_last_gw = TRUE;
1094  break;
1095  }
1096 
1097  if (!found_last_gw)
1098  j = -1; /* If search failed, we try the first gateway "again". */
1099 
1100  /* Now we look for the next gateway that could be used
1101  */
1102  while (++j < gate_top)
1103  {
1104  const struct gate_entry *gw = gate_list + j;
1105 
1106  if (/* sin_mask != IP_BCAST_ADDR && */ !is_on_LAN(gw->gate_ip))
1107  continue;
1108 
1109  if ((gw->mask & re->host_ip) != gw->subnet)
1110  continue;
1111 
1112  if (!LAN_start_lookup(gw->gate_ip))
1113  break; /* No room in ARP table, fail */
1114 
1115  re->gate_ip = gw->gate_ip; /* Ok, now we try this gateway */
1116  found_next_gw = TRUE;
1117  break;
1118  }
1119 
1120  /* No more gateways to try, hence lookup failed, kill entry
1121  */
1122  if (!found_next_gw)
1123  {
1124  if (i > route_first_pending)
1125  route_move_entry (i--, route_first_pending); /* fill hole */
1126  ++route_first_pending;
1127  }
1128  }
1129  }
1130 
1131  if (!check_dynamic_entries)
1132  return;
1133 
1134  /* Check our dynamic list for expired entries
1135  */
1136  for (i = ROUTE_FIRST_DYNAMIC; i < route_top_dynamic; i++)
1137  {
1138  re = route_list + i;
1139  if (LAN_lookup(re->gate_ip, NULL))
1140  continue; /* Still in ARP cache - ok */
1141 
1142  /* ARP entry expired. Remove from list.
1143  */
1144  if (i < --route_top_dynamic)
1145  {
1146  route_move_entry (i--, route_top_dynamic); /* fill hole */
1147  /* (i backed up a step so 'new' slot is rechecked) */
1148  re = route_list + i;
1149  re->flags &= ~ROUTE_FLG_USED;
1150  }
1151  }
1152 }
