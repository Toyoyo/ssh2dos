365 /*
366  * sock_wait_established()
367  * - Waits until connected or aborts if timeout etc.
368  *
369  * sock_wait_input()
370  * - Waits for received input on socket 's'.
371  * - May not be valid input for sock_gets().. check returned length.
372  *
373  * sock_tick()
374  * - Do tick and jump on abort.
375  *
376  * sock_wait_closed()
377  * - Close socket and wait until fully closed.
378  * Discards all received data.
379  *
380  * All these macros jump to label sock_err with contents of *statusptr
381  * set to
382  * 1 on closed normally.
383  * -1 on error, call sockerr(s) for cause.
384  *
385  */
386 W32_FUNC int W32_CALL _ip_delay0 (sock_type *s, int sec, UserHandler fn, int *statusptr);
387 W32_FUNC int W32_CALL _ip_delay1 (sock_type *s, int sec, UserHandler fn, int *statusptr);
388 W32_FUNC int W32_CALL _ip_delay2 (sock_type *s, int sec, UserHandler fn, int *statusptr);

391 #define sock_wait_established(s,seconds,fn,statusptr) \
392  do { \
393  if (_ip_delay0 (s,seconds,fn,statusptr)) \
394  goto sock_err; \
395  } while (0)
396 
397 #define sock_wait_input(s,seconds,fn,statusptr) \
398  do { \
399  if (_ip_delay1 (s,seconds,fn,statusptr)) \
400  goto sock_err; \
401  } while (0)
402 
403 #define sock_tick(s, statusptr) \
404  do { \
405  if (!tcp_tick(s)) { \
406  if (statusptr) *statusptr = -1; \
407  goto sock_err; \
408  } \
409  } while (0)

411 #define sock_wait_closed(s,seconds,fn,statusptr) \
412  do { \
413  if (_ip_delay2(s,seconds,fn,statusptr)) \
