180  * The get_file_version() fails under Win-Vista+ since files under
181  * '"%SystemRoot%\system32\drivers' are hidden from non-admin users.
182  * So this function is used instead. I assume the true file-version of

185 static BOOL get_npf_ver_from_registry (char *ret_ver, size_t ver_size)
186 {
187  char str[100];
188  BOOL rc = FALSE;
189  DWORD size = sizeof(str);
190  HKEY key = NULL;
191  LONG status;
192 
193  status = RegOpenKeyEx (HKEY_LOCAL_MACHINE,
194  "SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Uninstall\\WinPcapInst",
195  0, KEY_READ, &key);
196 
197  /* Note: WinPcap may be installed and working even though this key doesn't exist.
198  */
199  if (status != ERROR_SUCCESS)
200  goto fail;
201 
202  status = RegQueryValueEx (key, "DisplayVersion", NULL, NULL, (BYTE*)&str, &size);
203  if (status != ERROR_SUCCESS)
204  goto fail;
205 
206  _strlcpy (ret_ver, str, ver_size);
207  rc = TRUE;
208 
209 fail:
210  if (key)
211  RegCloseKey (key);
212  return (rc);
213 }

271  /*
272  * Retrieve NPF.sys version information from the file.
273  *
274  * \todo: Should maybe use an absolute path: "%SystemRoot%\system32\drivers\npf.sys" ?
275  * Or we can assume %SystemRoot% is always in %PATH?
276  *
277  * The get_file_version() fails under Win-Vista+ since files under
278  * '"%SystemRoot%\system32\drivers' are hidden from non-admin users.
279  * Hence try the "%SystemRoot%\sysnative\drivers" directory first
280  * (a virtual directory). If that fails (or on < Vista), try
281  * "%SystemRoot%\system32\drivers".
282  *
283  * Instead of checking "%SystemRoot%\sysnative\drivers" directly, MSDN
284  * would recommend we use:
285  * Wow64DisableWow64FsRedirection (...)
286  * get_file_version (driver_path,...)
287  * Wow64EnableWow64FsRedirection (....)
288  *
289  * But this can led to strange bugs.
290  *
291  * Refs: https://msdn.microsoft.com/en-us/library/windows/desktop/aa365743(v=vs.85).aspx
292  * https://blogs.msdn.microsoft.com/oldnewthing/20130321-00/?p=4883/
293  *
294  * If both attempts fails, simply try to get the version from Registry.
295  *
296  * Ref. Section "Registry and file system" at:
297  * http://en.wikipedia.org/wiki/WoW64
298  */
299  if (_watt_os_ver >= 0x0600)
300  rc = get_file_version (virtual_path, npf_drv_ver, sizeof(npf_drv_ver));
301 
302  if (!rc)
303  rc = get_file_version (driver_path, npf_drv_ver, sizeof(npf_drv_ver));
304 
305  WINPKT_TRACE ("get_file_version(): rc=%d, npf_drv_ver=\"%s\"\n", rc, npf_drv_ver);
306 
307  if (!rc)
308  {
309  rc = get_npf_ver_from_registry (npf_drv_ver, sizeof(npf_drv_ver));
310  WINPKT_TRACE ("get_npf_ver_from_registry(): rc=%d, npf_drv_ver=\"%s\"\n",
311  rc, npf_drv_ver);
312  }
313 
314  /* Populate the 'adapters_list' list.
315  */
316  rc = PopulateAdaptersInfoList();
317 
318  winpkt_trace_func = "PacketInitModule";
319  WINPKT_TRACE ("Known WinPcap adapters:\n");
320 
321  for (ai = PacketGetAdInfo(); ai; ai = ai->Next)
322  {
323  winpkt_trace_func = "PacketInitModule";
324 #if 0
325  WINPKT_TRACE ("%s, `%s'\n", ai->Name, ai->Description);
326 #else
327  WINPKT_TRACE ("%s\n", ai->Name);
328 #endif
329  }
330  WINPKT_TRACE ("rc %d\n", rc);
331 
332  /* Check if we loaded WanPacket.dll okay. We don't care if it fails (since
333  * it's not critical for Watt-32/Win. These 'p_' func-ptr are et in win_dll.c
334  */
335  use_wanpacket = (p_WanPacketSetBpfFilter && p_WanPacketOpenAdapter &&
336  p_WanPacketCloseAdapter && p_WanPacketSetBufferSize &&
337  p_WanPacketReceivePacket && p_WanPacketSetMinToCopy &&
338  p_WanPacketGetStats && p_WanPacketSetReadTimeout &&
339  p_WanPacketSetMode && p_WanPacketGetReadEvent &&
340  p_WanPacketTestAdapter);
341  return (rc);
342 }

377 
378  memset (&oid, 0, sizeof(oid));

403 static BOOL PacketSetReadEvt3xx (ADAPTER *AdapterObject)
404 {
405  DWORD BytesReturned;
406  char EventName[40];
407  WCHAR EventNameU[20];
408 
409  winpkt_trace_func = "PacketSetReadEvt3xx";
410 
411  /* retrieve the Unicode name of the shared event from the driver
412  */
413  memset (&EventNameU, 0, sizeof(EventNameU));
414  if (!DeviceIoControl(AdapterObject->hFile, pBIOCEVNAME, NULL, 0,
415  EventNameU, sizeof(EventNameU), &BytesReturned, NULL))
416  {
417  WINPKT_TRACE ("error retrieving the event-name from the kernel\n");
418  return (FALSE);
419  }
420 
421  /* this tells the terminal service to retrieve the event from the global
422  * namespace
423  */
424  SNPRINTF (EventName, sizeof(EventName), "Global\\%.13S", EventNameU);
425  WINPKT_TRACE (" event-name %s\n", EventName);
426 
427  /* open the shared event
428  */
429  AdapterObject->ReadEvent = CreateEventA (NULL, TRUE, FALSE, EventName);
430 
431  /* On Win-NT4 "Global\" is not automatically ignored: try to use
432  * simply the event name.
433  */
434  if (GetLastError() != ERROR_ALREADY_EXISTS)
435  {
436  if (AdapterObject->ReadEvent != INVALID_HANDLE_VALUE)
437  CloseHandle (AdapterObject->ReadEvent);
438 
439  /* open the shared event */
440  AdapterObject->ReadEvent = CreateEventA (NULL, TRUE, FALSE, EventName+7); /* skip "Global\" */
441  }
442 
443  if (AdapterObject->ReadEvent == INVALID_HANDLE_VALUE ||
444  GetLastError() != ERROR_ALREADY_EXISTS)
445  {
446  WINPKT_TRACE ("error retrieving the event from the kernel\n");
447  return (FALSE);
448  }
449 
450  AdapterObject->ReadTimeOut = 0; /* block until something received */
451  WINPKT_TRACE ("okay\n");
452  return (TRUE);
453 }

570 static ADAPTER *PacketOpenAdapterNPF (const char *AdapterName)
571 {
572  ADAPTER *lpAdapter;
573  BOOL QuerySStat, Result;
574  DWORD error, KeyRes;
575  SC_HANDLE svcHandle = NULL;
576  HKEY PathKey;
577  char SymbolicLink[128];
578  const char *sym_link;
579  SERVICE_STATUS SStat;
580  SC_HANDLE scmHandle, srvHandle;
581  size_t prefix_len;
582 
583  winpkt_trace_func = "PacketOpenAdapterNPF";
584  WINPKT_TRACE ("\n");
585 
586  set_char_pointers (AdapterName);
587 
588  scmHandle = OpenSCManager (NULL, NULL, GENERIC_READ);
589  if (!scmHandle)
590  {
591  error = GetLastError();
592  WINPKT_TRACE ("OpenSCManager failed; %s\n", win_strerror(error));
593  }
594  else
595  {
596  /* Check if the NPF registry key is already present.
597  * This means that the driver is already installed and that
598  * we don't need to call PacketInstallDriver
599  */
600  KeyRes = RegOpenKeyExA (HKEY_LOCAL_MACHINE, registry_location,
601  0, KEY_READ, &PathKey);
602  if (KeyRes != ERROR_SUCCESS)
603  {
604  Result = PacketInstallDriver (scmHandle, &svcHandle);
605  }
606  else
607  {
608  Result = TRUE;
609  RegCloseKey (PathKey);
610  }
611 
612  winpkt_trace_func = "PacketOpenAdapterNPF";
613  WINPKT_TRACE ("Service %s: \n", service_name);
614 
615  if (Result)
616  {
617  srvHandle = OpenServiceA (scmHandle, service_name,
618  SERVICE_START | SERVICE_QUERY_STATUS);
619  if (srvHandle)
620  {
621  QuerySStat = QueryServiceStatus (srvHandle, &SStat);
622 
623  WINPKT_TRACE ("state %s\n", list_lookup(SStat.dwCurrentState, serv_stat, DIM(serv_stat)));
624 
625  if (!QuerySStat || SStat.dwCurrentState != SERVICE_RUNNING)
626  {
627  WINPKT_TRACE ("Calling startservice\n");
628  if (!StartService(srvHandle, 0, NULL))
629  {
630  error = GetLastError();
631  if (error != ERROR_SERVICE_ALREADY_RUNNING &&
632  error != ERROR_ALREADY_EXISTS)
633  {
634  error = GetLastError();
635  CloseServiceHandle (scmHandle);
636  WINPKT_TRACE ("StartService failed; %s\n",
637  win_strerror(error));
638  SetLastError (error);
639  return (NULL);
640  }
641  }
642  }
643  CloseServiceHandle (srvHandle);
644  srvHandle = NULL;
645  }
646  else
647  {
648  error = GetLastError();
649  WINPKT_TRACE ("OpenService failed; %s\n", win_strerror(error));
650  SetLastError (error);
651  }
652  }
653  else /* Registry key not found or PacketInstallDriver() failed */
654  {
655  if (KeyRes != ERROR_SUCCESS)
656  Result = PacketInstallDriver (scmHandle, &svcHandle);
657  else Result = TRUE;
658 
659  winpkt_trace_func = "PacketOpenAdapterNPF";
660 
661  if (Result)
662  {
663  srvHandle = OpenServiceA (scmHandle, service_name, SERVICE_START);
664  if (srvHandle)
665  {
666  QuerySStat = QueryServiceStatus (srvHandle, &SStat);
667 
668  WINPKT_TRACE ("state: %s\n", list_lookup(SStat.dwCurrentState, serv_stat, DIM(serv_stat)));
669 
670  if (!QuerySStat || SStat.dwCurrentState != SERVICE_RUNNING)
671  {
672  WINPKT_TRACE ("Calling startservice\n");
673 
674  if (StartService (srvHandle, 0, NULL) == 0)
675  {
676  error = GetLastError();
677  if (error != ERROR_SERVICE_ALREADY_RUNNING &&
678  error != ERROR_ALREADY_EXISTS)
679  {
680  if (scmHandle)
681  CloseServiceHandle (scmHandle);
682  WINPKT_TRACE ("StartService failed; %s\n",
683  win_strerror(error));
684  SetLastError (error);
685  return (NULL);
686  }
687  }
688  }
689  CloseServiceHandle (srvHandle);
690  }
691  else
692  {
693  error = GetLastError();
694  WINPKT_TRACE ("OpenService failed; %s", win_strerror(error));
695  SetLastError (error);
696  }
697  }
698  }
699  } /* OpenSCManager() */
700 
701  if (scmHandle)
702  CloseServiceHandle (scmHandle);
703 
704  /* skip "\Device" to create a symlink name
705  */
706  prefix_len = strlen (DEVICE_PREFIX);
707  if (strnicmp(AdapterName,driver_prefix,prefix_len))
708  {
709  WINPKT_TRACE ("Unexpected prefix for adapter \"%s\". Expected prefix: \"%s\"\n",
710  AdapterName, driver_prefix);
711  return (NULL);
712  }
713 
714  lpAdapter = GlobalAllocPtr (GMEM_MOVEABLE | GMEM_ZEROINIT,
715  sizeof(*lpAdapter));
716  if (!lpAdapter)
717  {
718  WINPKT_TRACE ("Failed to allocate the adapter structure\n");
719  return (NULL);
720  }
721 
722  sym_link = AdapterName + prefix_len;
723  SNPRINTF (SymbolicLink, sizeof(SymbolicLink), "\\\\.\\%s", sym_link);
724 
725  WINPKT_TRACE ("Creating SymbolicLink: \"%s\".\n", SymbolicLink);
726 
727  /* try if it is possible to open the adapter immediately
728  */
729  lpAdapter->hFile = CreateFileA (SymbolicLink, GENERIC_WRITE | GENERIC_READ,
730  0, NULL, OPEN_EXISTING, 0, 0);
731 
732  if (lpAdapter->hFile != INVALID_HANDLE_VALUE)
733  {
734  if (!PacketSetReadEvt(lpAdapter))
735  {
736  error = GetLastError();
737  (void) GlobalFreePtr (lpAdapter);
738  WINPKT_TRACE ("failed; %s\n", win_strerror(error));
739  SetLastError (error);
740  return (NULL);
741  }
742 
743  /* Success!
744  */
745  PacketSetMaxLookaheadsize (lpAdapter);
746  winpkt_trace_func = "PacketOpenAdapterNPF";
747  return (lpAdapter);
748  }
749 
750  error = GetLastError();
751  (void) GlobalFreePtr (lpAdapter);
752  WINPKT_TRACE ("CreateFileA (%s) failed; \n%-49s%s\n",
753  SymbolicLink, "", win_strerror(error));
754  SetLastError (error);
755  return (NULL);
756 }
