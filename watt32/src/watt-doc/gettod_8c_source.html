108 uint64 FILETIME_to_unix_epoch (const FILETIME *ft)
109 {
110  uint64 res = (uint64) ft->dwHighDateTime << 32;
111 
112  res |= ft->dwLowDateTime;
113  res /= 10; /* from 100 nano-sec periods to usec */
114  res -= DELTA_EPOCH_IN_USEC; /* from Win epoch to Unix epoch */
115  return (res);

297 /*
298  * Return hardware time-of-day in microseconds.
299  * NOTE: hardware counter (lo) counts down from 65536 at a rate of
300  * 1.19318 MHz (4.77/4).
301  */
302 static __inline uint64 microsec_clock (void)
303 {
304  DWORD hi;
305  long lo;
306  uint64 rc;
307 
308  do
309  {
310  hi = PEEKL (0, BIOS_CLK);
311  lo = clockbits();
312  }
313  while (hi != PEEKL(0, BIOS_CLK)); /* tick count changed, try again */
314  lo = 0 - lo;
315  rc = ((uint64)hi << 16) + lo;
316  return (rc * U64_SUFFIX(4000000) / U64_SUFFIX(4770000));
317 }

320 /*
321  * Return CPU time-stamp counter in microseconds since epoch
322  * (1 Jan 1970 UTC) or from '*base'.
323  */
324 static uint64 tsc_microsec (const uint64 *base)
325 {
326  static uint64 time_ofs = U64_SUFFIX(-1);
327  static time_t start = 0;
328  uint64 rc, now;
329 
330  if (clocks_per_usec == 0)
331  return (0);
332 
333  if (time_ofs == U64_SUFFIX(-1))
334  {
335  start = time (NULL);
336  while (time(NULL) == start)
337  ENABLE();
338  time_ofs = GET_RDTSC();
339  }
340  if (base)
341  now = *base;
342  else now = GET_RDTSC();
343  rc = (now - time_ofs) / clocks_per_usec;
344  rc += U64_SUFFIX(1000000) * (start+1);
345  return (rc);
346 }

348 static void adjust_cpu_clock (const struct timeval *tv)
349 {
350  static DWORD last = 0;
351  DWORD tick = PEEKL (0, BIOS_CLK);
352 
353  if (((tick % 5) == 0) && last != tick) /* 260msec passed */
354  {
355  static double sample[4];
356  static int idx = 0;
357 
358  if (idx < DIM(sample))
359  {
360  struct timeval tv0;
361 
362  gettimeofday (&tv0, NULL);
363  sample[idx++] = timeval_diff (&tv0, tv);
364  }
365  else
366  {
367  double diff = 0.0;
368  int i, corr;
369 
370  for (i = 0; i < DIM(sample); i++)
371  diff += sample[i];
372  idx = 0;
373  diff = diff / DIM(sample);
374  corr = abs ((int)last-(int)tick) / 8;
375  corr = diff < 0.0 ? -corr : +corr;
376  clocks_per_usec += corr;
377 
378 #ifdef TEST_PROG
379  printf ("diff %.6f, corr %d\n", diff/1E6, corr);
380 #endif
381  }
382  last = tick;
383  }
