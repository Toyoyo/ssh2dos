61 static void print_reg (DWORD reg, const char *which)
62 {
63  BYTE a = LOBYTE (reg);
64  BYTE b = HIBYTE (reg);
65  BYTE c = LOBYTE (reg >> 16);
66  BYTE d = HIBYTE (reg >> 16);
67 
68  printf (" %s: %08lX, %c%c%c%c\n", which, reg,
69  isprint(a) ? a : '.',
70  isprint(b) ? b : '.',
71  isprint(c) ? c : '.',
72  isprint(d) ? d : '.');
73 }

88 __declspec(naked) static void msvc_cpuid (DWORD level, DWORD *_eax, DWORD *_ebx, DWORD *_ecx, DWORD *_edx)
89 {
90  __asm {
91  enter 0, 0
92  push ebx
93  push esi
94  mov eax, [ebp+8] /* EAX = level */
95  cpuid
96  mov esi, [ebp+12]
97  mov [esi], eax /* *_eax = EAX */
98  mov esi, [ebp+16]
99  mov [esi], ebx /* *_ebx = EBX */
100  mov esi, [ebp+20]
101  mov [esi], ecx /* *_ecx = ECX */
102  mov esi, [ebp+24]
103  mov [esi], edx /* *_edx = EDX */
104  pop esi
105  pop ebx
106  leave
107  ret
108  }
109 }

113 static unsigned int get_cpuid2 (int level, void *result)
114 {
115  unsigned int eax = 0, ebx = 0, ecx = 0, edx = 0;
116  unsigned int *res = (unsigned int*) result;
117  const char *cpuid_func;
118 
119 #if defined(__GNUC__)
120  cpuid_func = "__get_cpuid";
121  __get_cpuid (level, &eax, &ebx, &ecx, &edx);
122 #elif defined(_MSC_VER)
123  cpuid_func = "msvc_cpuid";
124  msvc_cpuid (level, &eax, &ebx, &ecx, &edx);
125 #endif
126 
127  if (trace_init() >= 1)
128  {
129  TRACE (1, "\nFrom %s (0x%08X):\n", cpuid_func, level);
130  print_reg (eax, "EAX");
131  print_reg (ebx, "EBX");
132  print_reg (ecx, "ECX");
133  print_reg (edx, "EDX");
134  }
135  res[0] = ebx;
136  res[1] = edx; /* Not a typo; we want EDX into 'res[1]' */
137  res[2] = ecx;
138  return (eax);
139 }

146  char features2 [12];

149  switch (type)

151  case 5:
152  if (model <= 2)
153  return ("Pentium");
154  if (model >= 4)
155  return ("PentiumMMX");
156 
157  case 6:
158  if (model == 1)
159  return ("PentiumPro");
160  if (model <= 6)
161 
162  return ("Pentium2");
163  if (model <= 13)
164  return ("Pentium3");
165 
166  if (model == 14 || model == 16)
167  return ("Core");
168 
169  if (model == 15 || model == 22)
170  return ("Core2");
171 
172  if (model == 17 || model == 23 || model == 29)
173  return ("Penryn");
174 
175  if (model == 25 || model == 37 || model == 44 || model == 47)
176  return ("Westmere");
177 
178  if (model == 26 || model == 30 || model == 31 || model == 46)
179  return ("Nehalem");
180 
181  if (model == 28 || model == 38 || model == 39 || model == 54 || model == 55)
182  return ("Atom");
183 
184  if (model == 42)
185  {
186  feat = ((int*) features) [2];
187  if (feat & 0x10000000)
188  return ("Sandybridge");
189  return ("Westmere");
190  }
191 
192  if (model == 43 || model == 45)
193  return ("Sandybridge");
194 
195  if (model == 58 || model == 62)
196  return ("Ivybridge");
197 
198  if (model == 60 || model == 63 || model == 69 || model == 70)
199  return ("Haswell");
200  break;
201 
202  case 15:
203  id_max = get_cpuid2 (0x80000000, features2) & 0xff;
204  if (id_max >= 1)
205  {
206  get_cpuid2 (0x80000001, features2);

209  return ("Netburst");

289 const char *cpu_get_model (void)
290 {
291  static char vendor_string[13];
292  char features[12];
293  int type, model, stepping;
294  DWORD id_max, fms;
295 
296  vendor_string[0] = '\0';
297  id_max = get_cpuid2 (0, vendor_string);
298  vendor_string[12] = '\0';
299 
300  TRACE (1, " id_max: %08lX.\n", id_max);
301 
302  fms = get_cpuid2 (1, features);
303 
304  type = (fms >> 8) & 0x0F;
305 /*if (type == 15) */
306  type |= (fms >> 16) & 0xFF0; /* set extended family */
307 
308  model = (fms >> 4) & 0x0F;
309 /*if (model == 15) */
310  model |= (fms >> 12) & 0xF0; /* set extended model */
311 
312  stepping = fms & 15;
313 
314  TRACE (1, "Found vendor_string: \"%s\".\n", vendor_string);
315  TRACE (1, "Found type: %d.\n", type);
316  TRACE (1, "Found model: %d.\n", model);
317  TRACE (1, "Found stepping: %d.\n", stepping);
318 
319  if (!strcmp(vendor_string,"GenuineIntel"))
320  return get_Intel_model (type, model, &features);
321 
322  if (!strcmp(vendor_string, "AuthenticAMD"))
323  return get_AMD_model (type, model);
324 
325  if (!strcmp(vendor_string, "CentaurHauls"))
326  return get_Centaur_model (type, model);
327 
328  if (vendor_string[0])
329  return (vendor_string);
330 
331  return (NULL);
332 }

334 const char *cpu_get_freq_info (void)
335 {
336  static char result [100];
337  char info[13];
338  DWORD id_max = get_cpuid2 (0, info);
339  unsigned int eax = 0, ebx = 0, ecx = 0;
340 
341  if (id_max < 0x16)
342  return (NULL);
343 
344  eax = get_cpuid2 (0x16, info);
345  ebx = ((unsigned int*) info) [1];
346  ecx = ((unsigned int*) info) [2];
347 
348  snprintf (result, sizeof(result), "Core base: %d, Core max: %d, Core bus: %d (MHz)",
349  eax & (1 << 15), ebx & (1 << 15), ecx & (1 << 15));
350  return (result);
351 }
