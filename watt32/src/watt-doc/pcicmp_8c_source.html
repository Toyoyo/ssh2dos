391 
392  if (block_icmp) /* application is handling ICMP on it's own; exit now */
393  return;
394 
395  len = in_GetHdrLen (ip);
396  icmp = (union ICMP_PKT*) ((BYTE*)ip + len);
397  len = intel16 (ip->length) - len;
398  for_me = _ip4_is_multihome_addr (intel(ip->destination));
399 
400  if (!for_me || broadcast) /* drop broadcast pings.. */
401  return;
402 
403  if (len < sizeof(icmp->info))
404  {
405  STAT (icmpstats.icps_tooshort++);
406  return;
407  }
408 
409  if (CHECKSUM(icmp,len) != 0xFFFF)

411  STAT (icmpstats.icps_checksum++);

413  return;
414  }
415 
416  type = icmp->unused.type;
417  code = icmp->unused.code;
418  orig_ip = &icmp->ip.ip;
419  i_orig = _ip4_is_local_addr (intel(orig_ip->source));
420 
421  if (type == ICMP_MASKREPLY)
422  {
423  if (!_do_mask_req)
424  return;
425  i_orig = TRUE;
426  }
427 
428  /* !! this needs work
429  */
430  if (!i_orig &&
431  (type != ICMP_ECHOREPLY && type != ICMP_ECHO &&
432  type != ICMP_IREQREPLY && type != ICMP_TSTAMP))
433  {
434  icmp_bogus (ip, type, NULL);
435  return;
436  }
437 
438  switch (type)
439  {
440  case ICMP_ECHOREPLY: /* check if we were waiting for it */
441  delta_time = set_timeout(0) - icmp->echo.identifier;
442  add_ping (intel(ip->source), delta_time, icmp->echo.index);
443  return;
444 
445  case ICMP_UNREACH:

447  {
448  UINT len_needed = 8 + in_GetHdrLen (orig_ip);
449  WORD next_mtu = 0;

452  icmp_print (1, msg, ip->source);
453 
454  if (orig_ip->proto == TCP_PROTO ||
455  orig_ip->proto == UDP_PROTO)
456  len_needed += 4; /* Need the src/dest port numbers */
457 
458  if (len >= len_needed)
459  {
460  if (code == ICMP_UNREACH_NEEDFRAG)
461  next_mtu = intel16 (icmp->needfrag.next_mtu);
462 
463 #if !defined(USE_UDP_ONLY)
464  if (orig_ip->proto == TCP_PROTO)
465  _tcp_cancel (orig_ip, ICMP_UNREACH, code, msg, &next_mtu);
466  else
467 #endif
468  if (orig_ip->proto == UDP_PROTO)
469  _udp_cancel (orig_ip, ICMP_UNREACH, code, msg, &next_mtu);
470 
472 #if defined(USE_BSD_API) && 0
473  else
474  _raw_cancel (orig_ip, ICMP_UNREACH, code, msg);
475 #endif
476  }
477  else
478  STAT (icmpstats.icps_tooshort++);
479  }
480  else
481  STAT (icmpstats.icps_badcode++);
482  return;
483 
484  case ICMP_SOURCEQUENCH:
485 #if !defined(USE_UDP_ONLY)
486  if (orig_ip->proto == TCP_PROTO)
487  {
488  msg = _LANG (icmp_type_str[type]);
489  icmp_print (1, msg, ip->source);

492 #endif
493  return;
494

497  icmp_redirect (icmp, ip, orig_ip, code);
498  else STAT (icmpstats.icps_badcode++);
499  return;
500 
501  case ICMP_ECHO:

503  icmp_echo_reply (ip, icmp, len);
504  return;
505

508  {

510  return;
511  }

514  {
515 #if !defined(USE_UDP_ONLY)
516  case TCP_PROTO:
517  msg = _LANG (icmp_exceed_str[0]);

520  break;
521 #endif

523  msg = _LANG (icmp_exceed_str[0]);
