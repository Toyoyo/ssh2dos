385 /*
386  * The following 2 routines are modified versions from KA9Q NOS
387  * by Phil Karn.
388  *
389  * clockbits() - Read low order bits of timer 0 (the TOD clock)
390  * This works only for the 8254 chips used in ATs and 386+.
391  *
392  * Old version:
393  * The timer 0 runs in mode 3 (square wave mode), counting down
394  * by 2s, twice for each cycle. So it is necessary to read back the
395  * OUTPUT pin to see which half of the cycle we're in. I.e., the OUTPUT
396  * pin forms the most significant bit of the count. Unfortunately,
397  * the 8253 in the PC/XT lacks a command to read the OUTPUT pin...
398  *
399  * Now simply:
400  * The timer 0 runs in mode 2 (rate generator), counting down.
401  * We don't compare the OUTPUT pin to see what cycle we're in.
402  * A bit lower precision, but a bit faster.
403  *
404  * WARNING!: This can cause serious trouble if anything else is also using
405  * timer interrupts. In that case use the above "userSuppliedTimerTick()".
406  * The reading of I/O registers gives bad precision under Windows.
407  */
408 DWORD clockbits (void)
409 {
410  static VOLATILE DWORD count; /* static because of PUSHF_CLI() */
411 
412 #if 1
413  /*
414  * The following version was suggested by Andrew Paulsen
415  * <andrew.paulsen@ndsu.nodak.edu>. Seem to work fine.
416  */
417  static BOOL started = FALSE;
418 
419  PUSHF_CLI();
420 
421  if (!started) /* Initialize once, it is periodic */
422  {
423  _outportb (0x43, 0x34); /* Set timer 0, mode 2 */
424  _outportb (0x40, 0x00); /* Count is decremented, then compared to 0 */
425  _outportb (0x40, 0x00); /* so set count to 0 to get 65536 */
426  started = TRUE;
427  }
428 
429  _outportb (0x43, 0x04); /* Latch present counter value command */
430  count = _inportb (0x40); /* LSB of count */
431  count |= _inportb (0x40) << 8; /* MSB of count */
432 
433  POPF();
434 
435  return (count);
436 
437 #else
438  static VOLATILE DWORD stat, count; /* static because of PUSHF_CLI() */
439 
440  do
441  {
442  PUSHF_CLI();
443  _outportb (0x43, 0xC2); /* Read Back Command (timer 0, 8254 only) */
444  stat = _inportb (0x40); /* get status of timer 0 */
445  count = _inportb (0x40); /* LSB of count */
446  count |= _inportb (0x40) << 8; /* MSB of count */
447  POPF();
448  }
449  while (stat & 0x40); /* reread if NULL COUNT bit set */
450 
451  stat = (stat & 0x80) << 8; /* Shift OUTPUT to MSB of 16-bit word */
452  count >>= 1; /* count = count/2 */
453  if (count == 0)
454  return (stat ^ 0x8000UL); /* return complement of OUTPUT bit */
455  return (count | stat); /* Combine OUTPUT with counter */
456 #endif
457 }

460 /*
461  * Return hardware time-of-day in milliseconds. Resolution is improved
462  * beyond 55 ms (the clock tick interval) by reading back the instantaneous
463  * 8254 counter value and combining it with the clock tick counter.
464  *
465  * Reading the 8254 is a bit tricky since a tick could occur asynchronously
466  * between the two reads in clockbits(). The tick counter is examined before
467  * and after the hardware counter is read. If the tick counter changes, try
468  * again.

470  * \note The hardware counter (lo) counts down from 65536 at a rate of
471  * 1.193181 MHz (4.77/4). System tick count (hi) counts up.
