105 void W32_CALL ReadEthersFile (void)
106 {
107  FILE *file;
108  char buf [2*MAX_HOSTLEN];
109 
110  if (!ethersFname[0])
111  return;
112 
113  if (!FOPEN_TXT(file, ethersFname))
114  {
115  netdb_warn (ethersFname);
116  return;
117  }
118 
119  while (fgets(buf,sizeof(buf),file))
120  {
121  struct ethent *e;
122  struct hostent *h;
123  char host_ip [MAX_HOSTLEN];
124  int save;
125  eth_address eth;
126 
127  if (!get_ether_entry(buf,&eth,host_ip,sizeof(host_ip)))
128  continue;
129 
130  if (num_entries == -1)
131  num_entries = 0;
132  num_entries++;
133 
134  save = called_from_resolve;
135  called_from_resolve = TRUE; /* prevent a DNS lookup */
136  h = gethostbyname (host_ip);
137  called_from_resolve = save;
138 
139  /* If 'h == NULL': means 'host_ip' contained a host-name that cannot be resolved
140  * at this moment. gethostbyname()+resolve() cannot be called at startup.
141  * If 'h != NULL': means 'host_ip' is simply an IPv4-address or a host-name (or alias)
142  * that is in '/etc/hosts' file.
143  */
144  if (!h)
145  {
146  CONSOLE_MSG (4, ("ReadEthersFile(): gethostbyname() failed\n"));
147  continue;
148  }
149  CONSOLE_MSG (4, ("\n"));
150 
151  e = calloc (sizeof(*e), 1);
152  if (!e)
153  break;
154 
155  e->ip_addr = ntohl (*(DWORD*)h->h_addr);
156  memcpy (&e->eth_addr, &eth, sizeof(e->eth_addr));
157 
158  /* Add this to the permanent ARP-cache.
159  */
160  _arp_cache_add (e->ip_addr, &e->eth_addr, FALSE);
161 
162  if (h->h_name)
163  {
164  e->name = strdup (h->h_name);
165  if (!e->name)
166  break;
167  }
168  e->next = eth0;
169  eth0 = e;
170  }
171 
172  FCLOSE (file);
173  RUNDOWN_ADD (end_ether_entries, 250);
174 }

176 /*
177  * Parse a string of text containing an ethernet address and
178  * hostname/IP-address and separate it into its component parts.
179  * E.g.
180  * in_buf -> "88-87-17-17-5a-3e 10.0.0.4"
181  * in_buf -> "E0-CA-94-3D-74-F0 printer"
182  */
183 #define MIN_LEN sizeof("0:0:0:0:0:0 a.b.c.d")
184 
185 static int get_ether_entry (char *in_buf, eth_address *e,
186  char *name, size_t name_max)
187 {
188  size_t len, i;
189  unsigned eth [sizeof(*e)];
190  BOOL ok, colon;
191  char *token = strltrim (in_buf);
192  char *ip_name;
193  char *tok_buf = NULL;
194 
195  if (*token == '#' || *token == ';' || *token == '\n' || strlen(token) < MIN_LEN)
196  return (0);
197 
198  colon = (token[2] == ':');
199 
200  ok = colon ? (sscanf(token, "%02x:%02x:%02x:%02x:%02x:%02x",
201  &eth[0], &eth[1], &eth[2],
202  &eth[3], &eth[4], &eth[5]) == DIM(eth)) :
203  (sscanf(token, "%02x-%02x-%02x-%02x-%02x-%02x",
204  &eth[0], &eth[1], &eth[2],
205  &eth[3], &eth[4], &eth[5]) == DIM(eth));
206  if (!ok)
207  {
208  CONSOLE_MSG (1, ("get_ether_entry(): sscanf() failed\n"));
209  return (0);
210  }
211 
212  token = strtok_r (token, " \t", &tok_buf);
213  ip_name = strtok_r (NULL, " #\t\n", &tok_buf);
214 
215  if (!token || !ip_name || (len = strlen(ip_name)) < 1 || len > name_max)
216  {
217  CONSOLE_MSG (1, ("get_ether_entry(): short line or malformed ip_name '%s'\n", ip_name));
218  return (0);
219  }
220 
221  ip_name = (char*) expand_var_str (ip_name);
222 
223  _strlcpy (name, ip_name, name_max);
224 
225  for (i = 0; i < sizeof(*e); i++)
226  ((BYTE*)e)[i] = eth[i];
227 
228  CONSOLE_MSG (4, ("get_ether_entry(): ip: %s, eth: %02X:%02X:%02X:%02X:%02X:%02X\n",
229  ip_name, eth[0],eth[1],eth[2],eth[3],eth[4],eth[5]));
230 
231  return (1);
232 }
