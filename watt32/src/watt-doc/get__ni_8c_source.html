113 int W32_CALL getnameinfo (const struct sockaddr *sa, socklen_t salen,
114  char *host, socklen_t hostlen,
115  char *serv, socklen_t servlen, int flags)
116 {
117  static BOOL init = FALSE;
118  struct afd *afd;
119  sa_family_t family = -1;
120  DWORD v4a;
121  WORD port;
122  char numserv[512];
123  char numaddr[512];
124  char *addr, *p;
125  int i;
126 
127  const struct servent *sp;
128  const struct hostent *hp;
129  const struct sockaddr_in *a4 = (const struct sockaddr_in *) sa;
130  const struct sockaddr_in6 *a6 = (const struct sockaddr_in6*) sa;
131 
132  if (!init)
133  afd_list_init (TRUE);
134  init = TRUE;
135 
136  SOCK_DEBUGF (("\ngetnameinfo: "));
137 
138  if (!sa)
139  return GI_RESULT (ENI_NOSOCKET);
140 
141  family = sa->sa_family;
142 
143  SOCK_DEBUGF (("name %s, serv %.10s, flags %04X",
144  family == AF_INET ? inet_ntoa(a4->sin_addr) :
145  family == AF_INET6 ? _inet6_ntoa(&a6->sin6_addr) : "??",
146  ((flags & NI_NUMERICSERV) || flags == 0) ? "<get>" : serv,
147  flags));
148 
149  for (i = 0; afd_list[i].a_af; i++)
150  if (afd_list[i].a_af == family)
151  {
152  afd = &afd_list[i];
153  goto found;
154  }
155  return GI_RESULT (ENI_FAMILY);
156 
157 found:
158  if (salen != afd->a_socklen)
159  return GI_RESULT (ENI_SALEN);
160 
161  port = ((struct sockinet*)sa)->si_port; /* network byte order */
162  addr = (char*) sa + afd->a_off;
163 
164  SOCK_ENTER_SCOPE();
165 
166  if (!serv || servlen == 0)
167  {
168  /* do nothing in this case.
169  * in case you are wondering if "&&" is more correct than
170  * "||" here: RFC2553 says that serv == NULL OR servlen == 0
171  * means that the caller does not want the result.
172  */
173  }
174  else
175  {
176  if (flags & NI_NUMERICSERV)
177  sp = NULL;
178  else sp = getservbyport (port, (flags & NI_DGRAM) ? "udp" : "tcp");
179  if (sp)
180  {
181  if ((int)strlen(sp->s_name) + 1 > servlen)
182  return GI_RESULT (ENI_MEMORY);
183  strcpy (serv, sp->s_name);
184  }
185  else
186  {
187  itoa (ntohs(port), numserv, 10);
188  if ((int)strlen(numserv) + 1 > servlen)
189  return GI_RESULT (ENI_MEMORY);
190  strcpy (serv, numserv);
191  }
192  }
193 
194  switch (sa->sa_family)
195  {
196  case AF_INET:
197  v4a = ntohl (((struct sockaddr_in*)sa)->sin_addr.s_addr);
198  if (IN_MULTICAST (v4a) || IN_EXPERIMENTAL (v4a))
199  flags |= NI_NUMERICHOST;
200  v4a >>= IN_CLASSA_NSHIFT;
201  if (v4a == 0)
202  flags |= NI_NUMERICHOST;
203  break;
204 
205  case AF_INET6:
206  switch (a6->sin6_addr.s6_addr[0])
207  {
208  case 0x00:
209  if (IN6_IS_ADDR_V4MAPPED(&a6->sin6_addr))
210  ;
211  else if (IN6_IS_ADDR_LOOPBACK(&a6->sin6_addr))
212  ;
213  else
214  flags |= NI_NUMERICHOST;
215  break;
216  default:
217  if (IN6_IS_ADDR_LINKLOCAL(&a6->sin6_addr))
218  flags |= NI_NUMERICHOST;
219  else if (IN6_IS_ADDR_MULTICAST(&a6->sin6_addr))
220  flags |= NI_NUMERICHOST;
221  break;
222  }
223  break;
224  }
225 
226  if (!host || hostlen == 0)
227  {
228  /* do nothing in this case.
229  * in case you are wondering if "&&" is more correct than
230  * "||" here: RFC2553 says that host == NULL OR hostlen == 0
231  * means that the caller does not want the result.
232  */
233  }
234  else if (flags & NI_NUMERICHOST)
235  {
236  /* NUMERICHOST and NAMEREQD conflicts with each other
237  */
238  if (flags & NI_NAMEREQD)
239  return GI_RESULT (ENI_NOHOSTNAME);
240 
241  if (inet_ntop (afd->a_af, addr, numaddr, sizeof(numaddr)) == NULL)
242  return GI_RESULT (ENI_SYSTEM);
243 
244  if ((int)strlen(numaddr) + 1 > hostlen)
245  return GI_RESULT (ENI_MEMORY);
246 
247  strcpy (host, numaddr);
248 
249  if (afd->a_af == AF_INET6 &&
250  (IN6_IS_ADDR_LINKLOCAL(addr) ||
251  IN6_IS_ADDR_MULTICAST(addr)) &&
252  ((struct sockaddr_in6*)sa)->sin6_scope_id)
253  {
254 #if !defined(ALWAYS_WITHSCOPE)
255  if (flags & NI_WITHSCOPEID)
256 #endif
257  {
258  char *ep = strchr (host, '\0');
259  int id = ((struct sockaddr_in6*)sa)->sin6_scope_id;
260  int scope = __scope_id_to_ascii (id); /* '1' ... N */
261 
262  *ep++ = SCOPE_DELIMITER;
263  *ep++ = scope ? scope : '?';
264  *ep = '\0';
265  }
266  }
267  }
268  else
269  {
270  hp = gethostbyaddr (addr, afd->a_addrlen, afd->a_af);
271  if (hp)
272  {
273  if (flags & NI_NOFQDN)
274  {
275  p = strchr (hp->h_name, '.');
276  if (p)
277  *p = '\0';
278  }
279  if ((int)strlen(hp->h_name) + 1 > hostlen)
280  return GI_RESULT (ENI_MEMORY);
281  strcpy (host, hp->h_name);
282  }
283  else
284  {
285  if (flags & NI_NAMEREQD)
286  return GI_RESULT (ENI_NOHOSTNAME);
287 
288  if (inet_ntop (afd->a_af, addr, numaddr, sizeof(numaddr)) == NULL)
289  return GI_RESULT (ENI_NOHOSTNAME);
290 
291  if ((int)strlen(numaddr) + 1 > hostlen)
292  return GI_RESULT (ENI_MEMORY);
293 
294  strcpy (host, numaddr);
295  }
296  }
297  ARGSUSED (a4);
298  return GI_RESULT (NI_SUCCESS);
299 }
