476 static BOOL WinDnsQueryCommon (WORD type, const void *what,
477  void *result, size_t size)
478 {
479  DNS_RECORD *rr, *dr = NULL;
480  DNS_STATUS rc;
481  BOOL found;
482  DWORD opt = DNS_QUERY_NO_WIRE_QUERY | /* query the cache only */
483  DNS_QUERY_NO_NETBT | /* no NetBT names */
484  DNS_QUERY_NO_HOSTS_FILE; /* no Winsock hosts file */
485 
486  from_windns = FALSE;
487 
488  if (!p_DnsQuery_A || !p_DnsFree)
489  return (FALSE);
490 
491  rc = (*p_DnsQuery_A) ((const char*)what, type, opt, NULL, &dr, NULL);
492 
493  CONSOLE_MSG (2, ("DnsQuery_A: type %d, dr %p: %s\n",
494  type, dr, win_strerror(rc)));
495 
496  if (rc != ERROR_SUCCESS || !dr)
497  return (FALSE);
498 
499  dom_ttl = dr->dwTtl;
500  found = FALSE;
501 
502  for (rr = dr; rr; rr = rr->pNext)
503  {
504  CONSOLE_MSG (2, ("RR-type: %d: ", rr->wType));
505 
506  /* Use only 1st A/AAAA record
507  */
508  if (rr->wType == DNS_TYPE_A && type == DNS_TYPE_A)
509  {
510  DWORD ip = ntohl (rr->Data.A.IpAddress);
511 
512  CONSOLE_MSG (2, ("A: %s, ttl %lus\n",
513  _inet_ntoa(NULL,ip), (u_long)rr->dwTtl));
514  if (!found)
515  *(DWORD*) result = ip;
516  found = TRUE;
517  }
518 
519 #if defined(USE_BSD_API) || defined(USE_IPV6)
520  else if (rr->wType == DNS_TYPE_AAAA && type == DNS_TYPE_AAAA)
521  {
522  const void *ip6 = &rr->Data.AAAA.Ip6Address.IP6Dword[0];
523 
524  CONSOLE_MSG (2, ("AAAA: %s\n", _inet6_ntoa(ip6)));
525  if (!found)
526  memcpy (result, ip6, size);
527  found = TRUE;
528  }
529 #endif
530 
531  else if (rr->wType == DNS_TYPE_PTR && type == DNS_TYPE_PTR)
532  {
533  _tcsncpy (result, dr->Data.PTR.pNameHost, size);
534  CONSOLE_MSG (2, ("PTR: %" TSTR2ASCII_FMT "\n", (const TCHAR*)result));
535  }
536  else if (rr->wType == DNS_TYPE_CNAME)
537  {
538 #ifdef UNICODE
539  const char *src = wstring_acp (dr->Data.CNAME.pNameHost);
540 #else
541  const char *src = dr->Data.CNAME.pNameHost;
542 #endif
543  _strlcpy (dom_cname, src, sizeof(dom_cname));
544  CONSOLE_MSG (2, ("CNAME: %s\n", dom_cname));
545  }
546  else
547  CONSOLE_MSG (2, ("\n"));
548  }
549  (*p_DnsFree) (dr, DnsFreeRecordList);
550  from_windns = found;
551  return (TRUE);
552 }
