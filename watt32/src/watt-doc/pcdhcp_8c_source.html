244 {
245  DWORD my_ip = 0UL;

248  DHCP_state == DHCP_state_RENEWING ||

254  _eth_get_hwtype (&dhcp_out.dh_htype, &dhcp_out.dh_hlen);
255 
256  dhcp_out.dh_op = BOOTP_REQUEST;

288 {
289  BYTE *opt, *start;
290 
291  exchange_id = set_timeout (0); /* random exchange ID */
292  opt = start = make_boot_header();
293  *opt++ = DHCP_OPT_MSG_TYPE;

298

300  *opt++ = 2;

302  opt += 2;

312  *opt++ = DHCP_OPT_END;

319 static int DHCP_request (BOOL renew)
320 {
321  BYTE *opt = make_boot_header();
322 
323  *opt++ = DHCP_OPT_MSG_TYPE;
324  *opt++ = 1;
325  *opt++ = DHCP_REQUEST;
326 
327  if (!renew)
328  {
329  *opt++ = DHCP_OPT_SRV_IDENTIFIER;
330  *opt++ = sizeof (dhcp_server);
331  *(DWORD*)opt = intel (dhcp_server);
332  opt += sizeof (dhcp_server);
333  *opt++ = DHCP_OPT_REQUESTED_IP_ADDR;
334  *opt++ = sizeof (my_ip_addr);
335  *(DWORD*)opt = intel (my_ip_addr);
336  opt += sizeof (my_ip_addr);
337  }
338 
339  /* Some DHCP-daemons require this tag in a REQUEST.
340  */
341  opt = put_hardware_opt (opt);
342 
343 /* ... don't know why we would ever renew a lease
344  * for the time remaining on our current lease...
345  */
346 #if 0
347  if (dhcp_iplease && dhcp_iplease < (DWORD)-1)
348  {
349  *opt++ = DHCP_OPT_IP_ADDR_LEASE_TIME;
350  *opt++ = sizeof (dhcp_iplease);
351  *(DWORD*)opt = intel (dhcp_iplease);
352  opt += sizeof (dhcp_iplease);
353  }
354 #endif
355 
356  if (user_class.data)
357  {
358  *opt++ = DHCP_OPT_USER_CLASS;
359  *opt++ = user_class.size;
360  memcpy (opt, user_class.data, user_class.size);
361  opt += user_class.size;
362  }
363 
364  if (vend_class.data)
365  {
366  *opt++ = DHCP_OPT_CLASS_ID;
367  *opt++ = vend_class.size;
368  memcpy (opt, vend_class.data, vend_class.size);
369  opt += vend_class.size;
370  }
371 
372  *opt++ = DHCP_OPT_END;
373  TRACE (("DHCP req: renew(%d) dst(%s)\n",
374  renew, INET_NTOA(sock->udp.hisaddr)));
375  return DHCP_SEND (opt);
376 }

383 static int DHCP_release_decline (int msg_type, const char *msg)
384 {
385  BYTE *opt;
386 
387  exchange_id = set_timeout (0); /* new exchange ID */
388  opt = make_boot_header();
389 
390  *opt++ = DHCP_OPT_MSG_TYPE;
391  *opt++ = 1;
392  *opt++ = msg_type;
393  *opt++ = DHCP_OPT_SRV_IDENTIFIER;
394  *opt++ = sizeof (dhcp_server);
395  *(DWORD*)opt = intel (dhcp_server);
396  opt += sizeof (dhcp_server);
397 
398  if (msg)
399  {
400  const BYTE *end = &dhcp_out.dh_opt [sizeof(dhcp_out.dh_opt)-1];
401  size_t len = strlen (msg);
402 
403  len = min (len, 255);
404  len = min (len, (size_t)(end-opt-3));
405  *opt++ = DHCP_OPT_MSG;
406  *opt++ = (BYTE)len;
407  memcpy (opt, msg, len);
408  opt += len;
409  }
410  *opt++ = DHCP_OPT_END;
411  return DHCP_SEND (opt);

456 static int DHCP_offer (const struct dhcp *in)
457 {
458  int len;
459  DWORD ip;
460  const BYTE *opt = (const BYTE*) &in->dh_opt[4];
461 
462  while (opt < in->dh_opt + sizeof(in->dh_opt))
463  {
464  switch (*opt)
465  {
466  case DHCP_OPT_PAD:
467  opt++;
468  continue;
469 
470  case DHCP_OPT_SUBNET_MASK:
471  sin_mask = intel (*(DWORD*)(opt+2));
472  TRACE (("Net-mask: %s\n", INET_NTOA(sin_mask)));
473  break;
474 
475  case DHCP_OPT_TIME_OFFSET:
476 #if defined(USE_DEBUG)
477  {
478  long ofs = *(long*)(opt+2);
479  TRACE (("Time-ofs: %.2fh\n", (double)ofs/3600.0));
480  }
481 #endif
482  break;
483 
484  case DHCP_OPT_ROUTERS_ON_SNET:
485  {
486  static BOOL gw_added = FALSE;
487 
488  if (!gw_added)
489  _arp_kill_gateways(); /* delete gateways from cfg-file */
490  gw_added = TRUE;
491  router = intel (*(DWORD*)(opt+2));
492  _arp_add_gateway (NULL, router);
493  TRACE (("Gateway: %s\n", INET_NTOA(router)));
494  }
495  break;
496 
497  case DHCP_OPT_DNS_SRV:
498  {
499  static BOOL dns_added = FALSE;
500 
501  for (len = 0; len < *(opt+1); len += sizeof(DWORD))
502  {
503  ip = intel (*(DWORD*)(opt+2+len));
504  if (!dns_added)
505  last_nameserver = 0; /* delete nameserver from cfg-file */
506 
507  /*
508  * !!fix-me: Add only first name-server cause resolve() doesn't
509  * handle multiple servers (in different nets) very well
510  */
511  if (len == 0)
512  {
513  nameserver = ip;
514  _add_server (&last_nameserver, def_nameservers,
515  DIM(def_nameservers), ip);
516  }
517  dns_added = TRUE;
518  TRACE (("DNS: %s\n", INET_NTOA(ip)));
519  }
520  }
521  break;
522 
523 #if defined(USE_BSD_API)
524  case DHCP_OPT_LOG_SRV:
525  if (!syslog_host_name[0] && /* not already set */
526  opt[1] % 4 == 0) /* length = n * 4 */
527  {
528  ip = intel (*(DWORD*)(opt+2)); /* select 1st host */
529  _strlcpy (syslog_host_name, _inet_ntoa(NULL,ip),
530  sizeof(syslog_host_name));
531  TRACE (("Syslog: %s\n", syslog_host_name));
532  }
533  break;
534 
535  case DHCP_OPT_NBIOS_NAME_SRV:
536  ip = intel (*(DWORD*)(opt+2));
537  TRACE (("WINS: %s\n", INET_NTOA(ip)));
539  break;
540 
541  case DHCP_OPT_NBIOS_NODE_TYPE:
542  TRACE (("NBT node: %02X\n", opt[2]));
543  break;
544 
545  case DHCP_OPT_NBIOS_SCOPE:
546  TRACE (("NBT scope: %.*s\n", opt[1], opt+2));
547  break;
548 #endif
549  case DHCP_OPT_HOST_NAME:
550  /* Don't use sethostname() because '*(opt+2)' is not a FQDN.
551  */
552  len = min (opt[1], sizeof(hostname));
553  memcpy (hostname, opt+2, len);
554  hostname[len] = '\0';
555  TRACE (("Host name: `%s'\n", hostname));
556  break;
557 
558  case DHCP_OPT_DOMAIN_NAME:
559  len = min (opt[1], sizeof(defaultdomain)-1);
560  setdomainname ((const char*)(opt+2), len+1);
561  TRACE (("Domain: `%s'\n", def_domain));
562  break;
563 
564  case DHCP_OPT_IP_DEFAULT_TTL:
565  case DHCP_OPT_TCP_DEFAULT_TTL:
566  _default_ttl = opt[2];
567  break;
568 
569  case DHCP_OPT_MSG_TYPE:
570  if (opt[2] == DHCP_OFFER)
571  got_offer = TRUE;
572  break;
573 
574  case DHCP_OPT_MSG:
575  outsn ((const char*)(opt+2), *(opt+1));
576  break;
577 
578  case DHCP_OPT_SRV_IDENTIFIER:
579  dhcp_server = intel (*(DWORD*)(opt+2));
580  TRACE (("Server: %s\n", INET_NTOA(dhcp_server)));
581  break;
582 
583  case DHCP_OPT_IP_ADDR_LEASE_TIME:
584  dhcp_iplease = intel (*(DWORD*)(opt+2));
585  TRACE (("IP lease: %s\n", period(dhcp_iplease)));
586  break;
587 
588  case DHCP_OPT_T1_VALUE:
589  dhcp_renewal = intel (*(DWORD*)(opt+2));
590  TRACE (("Renewal: %s\n", period(dhcp_renewal)));
591  break;
592 
593  case DHCP_OPT_T2_VALUE:
594  dhcp_rebind = intel (*(DWORD*)(opt+2));
595  TRACE (("Rebind: %s\n", period(dhcp_rebind)));
596  break;
597 
598  case DHCP_OPT_TCP_KEEPALIVE_INTERVAL:
599 #if !defined(USE_UDP_ONLY)
600  tcp_keep_intvl = intel (*(DWORD*)(opt+2));
601 #endif
602  break;
603 
604  case DHCP_OPT_OVERLOAD:
605  switch (opt[2])
606  {
607  case 1:
608  TRACE (("Overload: `dh_file' options\n"));
609  dhcp_options_add (in->dh_file, sizeof(in->dh_file));
610  break;
611  case 2:
612  TRACE (("Overload: `dh_sname' options\n"));
613  dhcp_options_add (in->dh_sname, sizeof(in->dh_sname));
614  break;
615  case 3:
616  TRACE (("Overload: `dh_file/dh_sname' options\n"));
617  dhcp_options_add (in->dh_file, sizeof(in->dh_file));
618  dhcp_options_add (in->dh_sname, sizeof(in->dh_sname));
619  break;
620  }
621  break;
622 
623 #if defined(USE_TFTP)
624  case DHCP_OPT_TFTP_SERVER:
625  {
626  const char *serv;
627  len = opt[1];
628  serv = tftp_set_server ((const char*)(opt+2), len);
629  TRACE (("TFTP-serv: `%s'\n", serv));
630  ARGSUSED (serv);
631  }
632  break;
633 
634  case DHCP_OPT_BOOT_FILENAME:
635  {
636  const char *file;
637  len = opt[1];
638  file = tftp_set_boot_fname ((const char*)(opt+2), len);
639  TRACE (("BOOT-file: `%s'\n", file));
640  ARGSUSED (file);
641  }
642  break;
643 #endif
644 
645  case DHCP_OPT_USER_CLASS: /* these options are ignored in replies */
646  case DHCP_OPT_CLASS_ID:
647  break;
648 
649  case DHCP_OPT_GRUB_MENU: /* GRUB loader config-file */
650  TRACE (("GRUB: %.*s\n", opt[1], opt+2));
651  break;
652 
653  case DHCP_OPT_ETHERNET_ENCAPSULATION:
654  TRACE (("Encap: %s\n", opt[2] ? "IEEE 802.3" : "Ethernet II"));
655  if (opt[2] != 0)
656  outsnl (_LANG("Only Ethernet II encapsulation supported"));
657  break;
658 
659  case DHCP_OPT_END:
660  TRACE (("got end-option\n"));
661  return (got_offer);
662 
663  default:
664  TRACE (("Ignoring option %d\n", *opt));
665  break;
666  }
667  opt += *(opt+1) + 2;
668  }
669 
670  if (extra_options.data)
671  {
672  struct dhcp ext;
673 
674  len = min (extra_options.size, sizeof(ext.dh_opt));
675  extra_options.data [len] = DHCP_OPT_END;
676  memcpy (ext.dh_opt, extra_options.data, len);
677  DHCP_offer (&ext);
678  }
679  return (got_offer);
680 }

818 {
