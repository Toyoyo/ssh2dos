156 /*
157  * Handling of dynamic loading and unloading of DLLs and their functions.
158  */
159 int load_dynamic_table (struct LoadTable *tab, int tab_size)
160 {
161  int i;
162 
163  for (i = 0; i < tab_size; tab++, i++)
164  {
165  HINSTANCE mod_handle;
166 
167  if (!_watt_use_bugtrap && !_tcsicmp(tab->mod_name,BUGTRAP_DLL))
168  continue;
169 
170  if (i > 0 && !_tcsicmp(tab->mod_name, (tab-1)->mod_name))
171  mod_handle = (tab-1)->mod_handle;
172  else mod_handle = LoadLibrary (tab->mod_name);
173 
174  if (mod_handle && mod_handle != INVALID_HANDLE_VALUE)
175  {
176  FARPROC addr = GetProcAddress (mod_handle, tab->func_name);
177 
178  if (!addr)
179  CONSOLE_MSG (4, ("Function \"%s\" not found in %" TSTR2ASCII_FMT ".\n",
180  tab->func_name, tab->mod_name));
181  *tab->func_addr = addr;
182  }
183  tab->mod_handle = mod_handle;
184 
185  CONSOLE_MSG (4, ("%2d: Module 0x%08lX/%" TSTR2ASCII_FMT ", func \"%s\" -> 0x%" ADDR_FMT ".\n",
186  i, (u_long)HandleToUlong(tab->mod_handle), tab->mod_name, tab->func_name,
187  ADDR_CAST(*tab->func_addr)));
188  }
189  return (i);
190 }

192 int unload_dynamic_table (struct LoadTable *tab, int tab_size)
193 {
194  int i;
195 
196  for (i = 0; i < tab_size; tab++, i++)
197  {
198  int m_unload = 0;
199  int f_unload = 0;
200 
201  if (tab->mod_handle && tab->mod_handle != INVALID_HANDLE_VALUE)
202  {
203  FreeLibrary (tab->mod_handle);
204  m_unload = 1;
205  }
206  tab->mod_handle = INVALID_HANDLE_VALUE;
207 
208  if (*tab->func_addr)
209  f_unload = 1;
210  *tab->func_addr = NULL;
211 
212  CONSOLE_MSG (4, ("%2d: function \"%s\" %s. Module \"%" TSTR2ASCII_FMT "\" %s.\n",
213  i, tab->func_name, f_unload ? "freed" : "not used",
214  tab->mod_name, m_unload ? "unloaded" : "not used"));
215  }
216  return (i);
217 }
