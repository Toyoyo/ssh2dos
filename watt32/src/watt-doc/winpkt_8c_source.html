800 static DWORD __stdcall winpcap_recv_thread (void *arg)
801 {
802  int rc = 0;
803 
804  while (1)
805  {
806  const struct bpf_hdr *bp;
807  BYTE *pkt, *pkt_end;
808  size_t pkt_len, cap_len, hdr_len;
809  UINT total_len, chunk;
810 
811  if (!_pkt_inf || _pkt_inf->stop_thread) /* this signals closure */
812  {
813  rc = 1;
814  break;
815  }
816 
817  pkt_len = _pkt_inf->npf_buf_size;
818  pkt = _pkt_inf->npf_buf;
819 
820  total_len = (*_pkt_inf->recv_op) (_pkt_inf->adapter, pkt, pkt_len);
821  if (total_len <= sizeof(*bp))
822  {
823  rc = 2;
824  break;
825  }
826 
827  ENTER_CRIT();
828 
829  for (pkt_end = pkt + total_len, chunk = 1;
830  pkt < pkt_end;
831  pkt += Packet_WORDALIGN(cap_len+hdr_len), chunk++)
832  {
833  struct pkt_ringbuf *q;
834 
835  q = &_pkt_inf->pkt_queue;
836  bp = (const struct bpf_hdr*) pkt;
837 
838  cap_len = bp->bh_caplen;
839  hdr_len = bp->bh_hdrlen;
840 
841  CONSOLE_MSG (2, ("winpcap_recv_thread(): total_len %d, "
842  "chunk %d, cap_len %d, hdr_len %d, in-idx %d\n",
843  total_len, chunk, (int)cap_len, (int)hdr_len,
844  q->in_index));
845 
846  num_rx_bytes += total_len - sizeof(*bp);
847  num_rx_pkt++;
848 
849  if (cap_len > ETH_MAX)
850  {
851  _pkt_inf->error = "Large size";
852  STAT (macstats.num_too_large++);
853  }
854  else if (pktq_in_index(q) == q->out_index) /* queue is full, drop it */
855  q->num_drop++;
856  else
857  {
858  struct pkt_rx_element *head;
859  int pad_len, buf_max;
860 
861  head = (struct pkt_rx_element*) pktq_in_buf (q);
862  memcpy (head->rx_buf, pkt + hdr_len, cap_len);
863 
864  /* The NPF.SYS timestamp is very unrealiable. So we simply use
865  * QueryPerformanceCounter() as it is now.
866  */
867  head->tstamp_put = win_get_perf_count();
868  head->rx_length = cap_len;
869 
870  /* zero-pad up to ETH_MAX (don't destroy marker at end)
871  */
872  buf_max = q->buf_size - RX_ELEMENT_HEAD_SIZE - 4;
873  pad_len = min (ETH_MAX, buf_max - cap_len);
874  if (pad_len > 0)
875  memset (&head->rx_buf[cap_len], 0, pad_len);
876  pktq_inc_in (q);
877  }
878  }
879  LEAVE_CRIT();
880  }
881 
882  CONSOLE_MSG (2, ("winpcap_recv_thread(): rc %d\n", rc));
883  fflush (stdout);
884  ARGSUSED (arg);
885  thr_stopped = TRUE;
886  return (rc);
887 }
