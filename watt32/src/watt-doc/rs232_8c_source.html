43  * ___trace2com() - Strings passed here are dumped to the set-up serial
44  * COM port. If the string is < 16 chars long, it will fit into the
45  * 16550 UARTs FIFO and will not cause any significant delay.
46  *
47  * No interrupts are used, so this will not interfere with anything
48  * else that might be happening.
49  */

51 {

53  char buf [256];

76  }

117 int trace2com_init (WORD port_addr, DWORD baud_rate)
118 {
119  BYTE Lsb, Msb;
120  WORD base;
121  DWORD div;
122 
123  /* Check/get UART address
124  */
125  if (port_addr < 1 || port_addr > 4)
126  return (0);
127 
128  base = trace2com_stdPorts [port_addr-1];
129 
130  /* See if the chip is actually there and ready
131  */
132  if (_inportb (base+IER_REG) == 0xFF) /* Nothing here */
133  return (0);
134 
135  if ((_inportb (base+IER_REG) & 0x0F) != 0x00)
136  return (0); /* UART is already in use by another program */
137 
138  _outportb (base+IER_REG, 0); /* disable all interrupts */
139  if (_inportb (base+IER_REG) != 0)
140  return (0); /* Whatever is here is not an UART */
141 
142  /* Set up UARTs registers
143  */
144  div = 115200UL / baud_rate;
145  Msb = div >> 8;
146  Lsb = (div << 8) >> 8;
147 
148  _outportb (base+LCR_REG, 0x80); /* Turn address latch on */
149  _outportb (base+BAUD_LSB_REG, Lsb);
150  _outportb (base+BAUD_MSB_REG, Msb);
151  _outportb (base+LCR_REG, 0x00); /* Turn address latch off (again) */
152 
153  _outportb (base+IER_REG, 0x00); /* No interrupts, we use polling */
154  _outportb (base+FIFO_REG, 0x01|0x08); /* Activate FIFO @ MODE 2 */
155  _outportb (base+LCR_REG , 0x03); /* 8, N, 1 */
156  _outportb (base+MCR_REG, 0x03); /* DTR + DTS on */
157 
158  /* Check chip type, gives us the FIFO size
159  */
160  _outportb (base + FIFO_REG, 0x11);
161  Lsb = _inportb (base + FIFO_REG);
162 
163  /* If we have a 16750 or 16550 the FIFO size is 16, otherwise we assume 1
164  */
165  trace2com_fifoSize_1 = ((Lsb & 0x20) != 0x20 ||
166  (Lsb & 0xC0) == 0xC0) ? 15 : 0;
167 
168  /* We are now officially open for business
169  */
170  trace2com_speed = baud_rate;
171  trace2com_base = base;
172 
173  SIO_TRACE ((_LANG("Watt-32 COM%d trace started"), port_addr));
174  return (1);
175 }
