52 /*
53  * Common structure pcb for internet protocol implementation.
54  * Here are stored pointers to local and foreign host table
55  * entries, local and foreign socket numbers, and pointers
56  * up (to a socket structure) and down (to a protocol-specific)
57  * control block.
58  */
59 LIST_HEAD(inpcbhead, inpcb);
60 
61 struct inpcb {
62  LIST_ENTRY(inpcb) inp_list; /* list for all PCBs of this proto */
63  LIST_ENTRY(inpcb) inp_hash; /* hash list */
64  struct inpcbinfo *inp_pcbinfo;
65  struct in_addr inp_faddr; /* foreign host table entry */
66  u_short inp_fport; /* foreign port */
67  struct in_addr inp_laddr; /* local host table entry */
68  u_short inp_lport; /* local port */
69  struct socket *inp_socket; /* back pointer to socket */
70  caddr_t inp_ppcb; /* pointer to per-protocol pcb */
71  struct route inp_route; /* placeholder for routing entry */
72  int inp_flags; /* generic IP/datagram flags */
73  struct ip inp_ip; /* header prototype; should have more */
74  struct mbuf *inp_options; /* IP options */
75  struct ip_moptions *inp_moptions; /* IP multicast options */
76 };
