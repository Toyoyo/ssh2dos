131 /*
132  * Do this at most once each hour:
133  *
134  * 1. Get our WAN ip (<wan-ip> below) by fetching:
135  * http://checkip.dyndns.com/ or http://ipdetect.dnspark.com/
136  *
137  * 2. Send HTTP request to members.dyndns.com:
138  * GET /nic/update?system=dyndns&hostname=<host-name>&myip=<wan-ip> HTTP/1.1
139  * Host: members.dyndns.com
140  * Authorization: Basic <user-name>:<password> base64 encoded
141  * Connection: close
142  *
143  * 3. Get result: "nohost" means <host-name> isn't registered in DynDNS.
144  * "noauth" means wrong <user-name> or <passord>.
145  * "notfqdn" means invalid <host-name> etc.
146  * "nochg <wan-ip>" means no update needed.
147  * "good <wan-ip>" means the update was acccepted.
148  *
149  * \todo: this should be rewritten as a non-blocking Watt-32 daemon.
150  * Use daemon_add (dynip_exec).
151  */

157

159  return (0);

429 static int base64encode (const char *in, char *out, size_t out_len)
430 {
431  int c1, c2, c3;
432  static char basis_64[] = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"\
433  "abcdefghijklmnopqrstuvwxyz0123456789+/";
434  while (*in)
435  {
436  c1 = *in++;
437  if (*in == '\0')
438  c2 = c3 = 0;
439  else
440  {
441  c2 = *in++;
442  if (*in == '\0')
443  c3 = 0;
444  else c3 = *in++;
445  }
446  out_len -= 4;
447  if (out_len <= 4)
448  return (-1);
449 
450  *out++ = basis_64 [c1 >> 2];
451  *out++ = basis_64 [((c1 & 0x3) << 4) | ((c2 & 0xF0) >> 4)];
452  if (c2 == 0 && c3 == 0)
453  {
454  *out++ = '=';
455  *out++ = '=';
456  }
457  else if (c3 == 0)
458  {
459  *out++ = basis_64 [((c2 & 0xF) << 2) | ((c3 & 0xC0) >> 6)];
460  *out++ = '=';
461  }
462  else
463  {
464  *out++ = basis_64 [((c2 & 0xF) << 2) | ((c3 & 0xC0) >> 6)];
465  *out++ = basis_64 [c3 & 0x3F];
466  }
467  }
468  *out = '\0';
469  return (0);
470 }

477 static BOOL get_header (void *sock, long *cont_len_ptr)
478 {
479  int len, response;
480  long content_length;
481  size_t i;
482  char buf [512];
483  char *s;
484 
485  *cont_len_ptr = 0L;
486 
487  len = sock_gets (sock, (BYTE*)buf, sizeof(buf));
488  if (len <= 0)
489  {
490  TRACE (1, ("EOF from server\n"));
491  return (FALSE);
492  }
493  TRACE (1, ("HTTP: `%.*s'\r\n", len, buf));
494 
495  if (!STARTS_WITH(buf,HTTPVER_10) && /* no HTTP/1.[01]? */
496  !STARTS_WITH(buf,HTTPVER_11))
497  {
498  TRACE (1, ("Not a HTTP/1.[01] server\n"));
499  return (FALSE);
500  }
501 
502  s = buf + sizeof(HTTPVER_10)-1;
503  i = strspn (s, " \t");
504  if (i == 0) /* no whitespace? */
505  {
506  TRACE (1, ("Malformed HTTP/1.[01] line\n"));
507  return (FALSE);
508  }
509 
510  s += i;
511  response = 500;
512  sscanf (s, "%3d", &response);
513  if (response == 401)
514  {
515  TRACE (1, ("Authorisation failed\n"));
516  return (FALSE);
517  }
518  if (response != 200 && response != 301 &&
519  response != 302 && response != 304)
520  {
521  TRACE (1, ("Unexpected response %s\n", s));
522  return (FALSE);
523  }
524 
525  /* Eat up the other header lines here.
526  * Set to LONG_MAX, in case no "Content-length" header.
527  */
528  content_length = LONG_MAX;
529 
530  while ((len = sock_gets(sock, (BYTE*)buf, sizeof(buf))) > 0)
531  {
532  TRACE (1, ("HTTP: `%.*s'\r\n", len, buf));
533  s = buf;
534  if (STARTS_WITH(s,"Content-Length:"))
535  {
536  s += sizeof("Content-Length:") - 1;
537  content_length = atol (s);
538  }
539 #if 0
540  else if (STARTS_WITH(s,"Transfer-encoding: chunked"))
541  {
542  TRACE (1, ("Chunked encoding not supported\n"));
543  return (FALSE);
544  }
545 #endif
546  else if (STARTS_WITH(buf, "Location:"))
547  {
548  if (response == 301 || response == 302)
549  TRACE (1, ("Location at %s\n", buf));
550  }
551  else if (strchr(" \t", buf[0]))
552  TRACE (1, ("Warning: continuation line encountered\n"));
553  }
554  *cont_len_ptr = content_length;
555  return (TRUE);
556 }

561 static int get_url (const char *host, int port, const char *path,
562  const char *user_pass)
563 {
564  DWORD addr;
565  int status = 0;
566  int length;
567  long content_length = 0L;
568  char req_buf [512], *p;
569  sock_type *sock;
570 
571  if (!host)
572  {
573  TRACE (1, ("get_url: NULL host\n"));
574  return (-1);
575  }
576 
577  addr = resolve (host);
578  if (!addr)
579  {
580  TRACE (1, ("%s\n", dom_strerror(dom_errno)));
581  return (-1);
582  }
583 
584  sock = malloc (sizeof(*sock));
585  if (!sock)
586  {
587  puts ("get_url: No memory");
588  return (-1);
589  }
590 
591  if (!tcp_open(&sock->tcp, 0, addr, port, NULL))
592  {
593  printf ("Cannot connect to `%s'\n", host);
594  free (sock);
595  return (-1);
596  }
597 
598  length = 0;
599 
600  sock_wait_established (sock, 10, NULL, &status);
601  if (!tcp_tick(sock)) /* in case they sent reset */
602  {
603  status = -1;
604  goto sock_err;
605  }
606 
607  p = req_buf;
608  p += _SNPRINTF (BUF(req_buf),
609  "GET %s %s\r\n"
610  "Host: %s\r\n"
611  "User-Agent: %s\r\n",
612  path, HTTPVER, host, USER_AGENT);
613  if (user_pass)
614  {
615  char encoded_auth [256];
616 
617  if (base64encode(user_pass, encoded_auth, sizeof(encoded_auth)) < 0)
618  goto sock_err;
619  p += sprintf (p, "Authorization: Basic %s\r\n", encoded_auth);
620  }
621 
622  p += sprintf (p, "Connection: close\r\n\r\n");
623 
624  TRACE (2, ("Sent: %s\n", req_buf));
625 
626  sock_write (sock, (BYTE*)req_buf, (int)(p - req_buf));
627  sock_wait_input (sock, 10, NULL, &status);
628 
629  if (get_header (sock, &content_length))
630  TRACE (2, ("Content-length %ld, remaining %u\n",
631  content_length, sock_dataready(sock)));
632 
633  if (content_length >= 0L)
634  {
635  while (sock_dataready(sock) > 0)
636  {
637  int i = sock_gets (sock, (BYTE*)(resp_buf+length),
638  sizeof(resp_buf)-1-length);
639  TRACE (1, ("%s", resp_buf+length));
640  length += i;
641  if (length > SIZEOF(resp_buf))
642  {
643  length = SIZEOF(resp_buf);
644  break;
645  }
646  }
647  if (content_length != LONG_MAX && length != content_length)
648  TRACE (1, ("Warning, actual length = %d, content length = %ld\n",
649  length, content_length));
650  }
651 
652 sock_err:
653  sock_abort (sock); /* Kill 'sock' from the '_tcp_allsocs' list. */
654 
655  if (status == -1)
656  TRACE (1, ("get_url(): %s, %s, ", host, sockerr(sock)));
657  TRACE (2, ("got %d bytes\n", length));
658  free (sock);
659  return (length);
660 }
